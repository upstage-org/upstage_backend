set -e

# --- INPUT chain: remove unneeded/duplicate rules (from your existing config) ---
# Add duplicate/unwanted LOG rules and such here for the ports used in UpStage.

# --- INPUT chain: ensure public ports accepted (host-facing) ---
for port in 22 80 443 1884 9002; do
  iptables -C INPUT -p tcp --dport "$port" -j ACCEPT 2>/dev/null || iptables -I INPUT 1 -p tcp --dport "$port" -j ACCEPT
done

# --- DOCKER-USER chain: flush and rebuild (controls forwarded traffic to containers) ---
iptables -F DOCKER-USER

# 1. Established/related always allowed
iptables -A DOCKER-USER -m conntrack --ctstate RELATED,ESTABLISHED -j ACCEPT

# 2. Internal Docker/bridge ranges: RETURN so Docker FORWARD handling continues (all bridge ranges)
iptables -A DOCKER-USER -s 10.0.0.0/8 -j RETURN

# May appear as 172.17,18,20, etc. docker network ls and docker network inspect all of your existing bridges, and add ranges here.
iptables -A DOCKER-USER -s 172.16.0.0/12 -j RETURN 
iptables -A DOCKER-USER -s 192.168.0.0/16 -j RETURN

# 3. Restrict 5433 (Postgres) and 27018 (Mongo) to internal IP only
iptables -A DOCKER-USER -p tcp -s 10.116.0.4 --dport 5433 -j ACCEPT
iptables -A DOCKER-USER -p tcp -s 10.116.0.4 --dport 27018 -j ACCEPT
iptables -A DOCKER-USER -p tcp --dport 5433 -j DROP
iptables -A DOCKER-USER -p tcp --dport 27018 -j DROP

# 4. Public access to MQTT ports (forwarded to mosquitto container)
iptables -A DOCKER-USER -p tcp --dport 1884 -j ACCEPT
iptables -A DOCKER-USER -p tcp --dport 9002 -j ACCEPT

# 5. Public access to SSH and HTTP/HTTPS (in case of port-forward or host services)
iptables -A DOCKER-USER -p tcp --dport 22 -j ACCEPT
iptables -A DOCKER-USER -p tcp --dport 80 -j ACCEPT
iptables -A DOCKER-USER -p tcp --dport 443 -j ACCEPT

# 6. Default: RETURN to FORWARD chain (Docker and rest of stack continue)
iptables -A DOCKER-USER -j RETURN

echo "Done. Public: 22,80,443,1884,9002. Restricted to 10.116.0.4: 5433, 27018. Internal ranges 10/8, 172.16/12, 192.168/16 RETURN."
